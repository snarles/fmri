plotit <- function(d, k) {
  qs <- qs_par(d, k)
  vals <- get_q_ints(qs, k)
  plot(qs, type = "l")
  title(paste0("d=", d, " k=", k, 
              " a=", formatC(vals[2], 2), 
              "  b=", formatC(vals[3], 2)))
}
